# keycloak


## env vars

```sh
KEYCLOAK_VERSION=26.1.4
KEYCLOAK_ADMIN_USERNAME=keycloak
KEYCLOAK_ADMIN_PASSWORD=..

KEYCLOAK_DATABASE_HOST=postgres
KEYCLOAK_DATABASE_PORT=5432
KEYCLOAK_DATABASE_NAME=keycloak
KEYCLOAK_DATABASE_USER=...
KEYCLOAK_DATABASE_PASSWORD=...
KEYCLOAK_DATABASE_SCHEMA=public

DB_URL=jdbc:postgresql://${KEYCLOAK_DATABASE_HOST}:${KEYCLOAK_DATABASE_PORT}/${KEYCLOAK_DATABASE_NAME}
DB_USER=${KEYCLOAK_DATABASE_USER}
DB_PASSWORD=${KEYCLOAK_DATABASE_PASSWORD}
```

## create database

```sql
CREATE DATABASE keycloak;   
CREATE USER keycloak WITH PASSWORD '...';
GRANT ALL PRIVILEGES ON DATABASE keycloak TO keycloak;
GRANT CREATE ON SCHEMA public  TO keycloak;
```
